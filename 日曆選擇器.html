日曆選擇器

FullCalendar 官網

程式碼範例

                // 初始化 FullCalendar
            var calendar = new FullCalendar.Calendar(calendarEl, {
                initialView: 'dayGridMonth',
                headerToolbar: {
                    left: 'prev,next today',
                    center: 'title',
                    right: 'dayGridMonth,timeGridWeek,timeGridDay'
                },
                events: [
                    {
                        title: '事件 1',
                        start: '2024-10-01'
                    },
                    {
                        title: '事件 2',
                        start: '2024-10-07',
                        end: '2024-10-10'
                    }
                ]
            });

            // 渲染日曆
            calendar.render();
